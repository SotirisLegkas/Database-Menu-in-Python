drop table Brands,Customers,Dealers,Models,Options,vehicle,Orders

CREATE TABLE Brands
(Brand_ID NUMERIC(10) NOT NULL,
Brand_Name VARCHAR(20) NOT NULL,
Brand_Adress VARCHAR(20) NOT NULL,
Brand_Phone NUMERIC(10) NOT NULL,
PRIMARY KEY(Brand_ID));

CREATE TABLE Customers
(Customer_ID NUMERIC(10) NOT NULL,
Customer_Name VARCHAR(20) NOT NULL,
Customer_Phone NUMERIC(10) NOT NULL,
Customer_Address VARCHAR(20),
Customer_Gender VARCHAR(10),
PRIMARY KEY(Customer_ID));

CREATE TABLE Dealers
(Dealer_ID NUMERIC(10) NOT NULL,
Dealer_Name VARCHAR(20) NOT NULL,
Dealer_Phone NUMERIC(10) NOT NULL,
Dealer_Address VARCHAR(20) NOT NULL,
PRIMARY KEY(Dealer_ID));

CREATE TABLE Models
(Model_ID NUMERIC(10) NOT NULL,
Model_Name VARCHAR(20) NOT NULL,
Model_Year NUMERIC(4) NOT NULL,
Brand_ID NUMERIC(10) NOT NULL REFERENCES Brands(Brand_ID),
PRIMARY KEY(Model_ID));

CREATE TABLE Options
(Option_ID NUMERIC(10) NOT NULL,
Edition VARCHAR(10) NOT NULL,
Colour VARCHAR(10) NOT NULL,
Price NUMERIC(7) NOT NULL,
Model_ID NUMERIC(10) NOT NULL REFERENCES Models(Model_ID),
PRIMARY KEY(Option_ID));

CREATE TABLE Vehicle
(VIN NUMERIC(10) NOT NULL,
Brand_ID NUMERIC(10) NOT NULL REFERENCES Brands(Brand_ID),
Model_ID NUMERIC(10) NOT NULL REFERENCES Models(Model_ID),
Option_ID NUMERIC(10) NOT NULL REFERENCES Options(Option_ID),
PRIMARY KEY(VIN));

CREATE TABLE Orders
(Order_ID NUMERIC(10) NOT NULL,
Order_Date NUMERIC(8) NOT NULL,
Order_Completion VARCHAR(8) NOT NULL,
Order_Price NUMERIC(7) NOT NULL,
Dealer_ID NUMERIC(10) NOT NULL REFERENCES Dealers(Dealer_ID),
Customer_ID NUMERIC(10) NOT NULL REFERENCES Customers(Customer_ID),
VIN NUMERIC(10) NOT NULL REFERENCES Vehicle(VIN),
PRIMARY KEY(Order_ID));
